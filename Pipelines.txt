pipeline {
    agent any

    environment {
        URL_TEST="https://www.google.com" // URL a comprobar
    }

    stages{
        stage('Monitoring URL') {
            steps {
                script{
                    print("-------------------------------------------------------------------------------------------------------------------")
                    print("-------------------------------------------------------------------------------------------------------------------")
                    print("Monitoring: $URL_TEST")
                    print("-------------------------------------------------------------------------------------------------------------------")
                    print("-------------------------------------------------------------------------------------------------------------------")
                    sh 'curl --head $URL_TEST | grep HTTP'
                }
            }
        }
    }  
}
